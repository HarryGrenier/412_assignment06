399  std::default_random_engine generator(std::random_device{}());
400  std::uniform_int_distribution<int> distributionRow(data->startRow, data->endRow - 1);
401  std::uniform_int_distribution<int> distributionCol(0, data->outputImage->width - 1);
402 
403  while (true) {

405  int centerCol = distributionCol(generator); // Random column
406 
407  std::set<int> uniqueRegionIndices;
408  for (int i = -windowSize / 2; i <= windowSize / 2; ++i) {
409  for (int j = -windowSize / 2; j <= windowSize / 2; ++j) {
410  unsigned int targetRow = centerRow + i;
411  unsigned int targetCol = centerCol + j;
412  if (targetRow >= data->startRow && targetRow < data->endRow && targetCol >= 0 && targetCol < data->outputImage->width) {
413  int regionRow = targetRow / (data->outputImage->height / GRID_ROWS);
414  int regionCol = targetCol / (data->outputImage->width / GRID_COLS);
415  uniqueRegionIndices.insert(regionRow * GRID_COLS + regionCol);
