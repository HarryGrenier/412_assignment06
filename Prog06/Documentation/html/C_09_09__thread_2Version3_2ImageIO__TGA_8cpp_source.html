29 // Used because TGA stores the RGB data in reverse order (BGR)

31 void swapRGB_(unsigned char* theData, unsigned int height, unsigned int width)
32 {
33  unsigned int numBytes = height*width;
34 
35  for(unsigned int k = 0; k < numBytes; k++)
36  {
37  unsigned char tmp = theData[k*3+2];
38  theData[k*3+2] = theData[k*3];
39  theData[k*3] = tmp;
40  }
41 }

77  char head[18] ;
78  fread( head, sizeof(char), 18, tga_in ) ;
79  /* Get the size of the image */
80  unsigned int imgWidth = ((unsigned int)head[12]&0xFF) | (unsigned int)head[13]*256;
81  unsigned int imgHeight = ((unsigned int)head[14]&0xFF) | (unsigned int)head[15]*256;
82  unsigned int numBytes = imgWidth * imgHeight;
83 
84  ImageType imgType;
85  //unsigned short maxVal;
86  //unsigned int bytesPerPixel;
87  //unsigned int bytesPerRow;

91  //maxVal = 255;

93  //bytesPerRow = 4*imgWidth;

96  {

98  //maxVal = 255;
99  //bytesPerPixel = 1;
100  //bytesPerRow = imgWidth;
101  }
102  else
103  {
104  printf("Unsuported TGA image: ");
105  printf("Its type is %d and it has %d bits per pixel.\n", head[2], head[16]);
106  printf("The image must be uncompressed while having 8 or 24 bits per pixel.\n");
107  fclose(tga_in);
108  exit(12);
109  }
110 
111  RasterImage* image = new RasterImage(imgWidth, imgHeight, imgType);
112  unsigned char* data = (unsigned char*) image->raster;
113 
114  //--------------------------------
115  // Read the pixel data
116  //--------------------------------
117 
118  // Case of a color image
119  //------------------------
120  //if (image.type == kTGA_COLOR) *****************************
121  if(image->type == RGBA32_RASTER)
122  {
123  // First check if the image is mirrored vertically (a bit setting in the header)
124  if(head[17]&0x20)
125  {
126  unsigned char* ptr = data + numBytes*4 - image->bytesPerRow;
127  for(unsigned int i = 0; i < image->height; i++)

146  }

148  }

187 {
188  //--------------------------------
189  // open TARGA output file
190  //--------------------------------
191  FILE* tga_out = fopen(filePath, "wb" );
192  if (tga_out == NULL)
193  {
194  printf("Cannot create image file %s \n", filePath);
195  return kCannotOpenWrite;
196  }
197 
198  // Yes, I know that I tell you over and over that cascading if-else tests
199  // are bad style when testing an integral value, but here only two values
200  // are supported. If I ever add one more I'll use a switch, I promise.
201 
202  //------------------------------
203  // Case of a color image
204  //------------------------------
205  if (image->type == RGBA32_RASTER)
206  {
207  //--------------------------------
208  // create the header (TARGA file)
209  //--------------------------------
210  char head[18] ;
211  head[0] = 0 ; // ID field length.
212  head[1] = 0 ; // Color map type.
213  head[2] = 2 ; // Image type: true color, uncompressed.
214  head[3] = head[4] = 0 ; // First color map entry.
215  head[5] = head[6] = 0 ; // Color map lenght.
216  head[7] = 0 ; // Color map entry size.
217  head[8] = head[9] = 0 ; // Image X origin.
218  head[10] = head[11] = 0 ; // Image Y origin.
219  head[13] = (char) (image->width >> 8) ; // Image width.
220  head[12] = (char) (image->width & 0x0FF) ;
221  head[15] = (char) (image->height >> 8) ; // Image height.
222  head[14] = (char) (image->height & 0x0FF) ;
223  head[16] = 24 ; // Bits per pixel.
224  head[17] = 0 ; // Image descriptor bits ;
225  fwrite(head, sizeof(char), 18, tga_out );
226 
227  unsigned char* data = (unsigned char*) image->raster;
228  for(unsigned int i = 0; i < image->height; i++)
229  {
230  unsigned long offset = i*4*image->width;
231  for(unsigned int j = 0; j < image->width; j++)
232  {
233  fwrite(&data[offset+2], sizeof(char), 1, tga_out);
234  fwrite(&data[offset+1], sizeof(char), 1, tga_out);
235  fwrite(&data[offset], sizeof(char), 1, tga_out);
236  offset+=4;
237  }
238  }
239 
240  fclose( tga_out ) ;
241  }
242  //------------------------------
243  // Case of a gray-level image
244  //------------------------------
245  else if (image->type == GRAY_RASTER)
246  {
247  //--------------------------------
248  // create the header (TARGA file)
249  //--------------------------------
250  char head[18] ;
251  head[0] = 0 ; // ID field length.
252  head[1] = 0 ; // Color map type.
253  head[2] = 3 ; // Image type: gray-level, uncompressed.
254  head[3] = head[4] = 0 ; // First color map entry.
255  head[5] = head[6] = 0 ; // Color map lenght.
256  head[7] = 0 ; // Color map entry size.
257  head[8] = head[9] = 0 ; // Image X origin.
258  head[10] = head[11] = 0 ; // Image Y origin.

261  head[15] = (char) (image->height >> 8) ; // Image height.
